class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> mappa1 = new HashMap<>();
        int[]arr= new int[2];
        for(int i=0; i<nums.length; i++){
            int diff= target-nums[i];
            if(mappa1.containsKey(diff)){
                arr[1]=i;
                arr[0]=mappa1.get(diff);

            }
            mappa1.put(nums[i],i);
        }
        return arr;
    }
}
